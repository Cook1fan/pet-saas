# 状态管理

## Pinia Store 设计

### 用户状态 (user.ts)

```typescript
// stores/user.ts
import { defineStore } from 'pinia'
import { login as loginApi, getCurrentUser } from '@/api/auth'

interface UserState {
  token: string | null
  userInfo: any | null
  tenantId: number | null
  role: 'platform_admin' | 'shop_admin' | 'shop_staff' | null
}

export const useUserStore = defineStore('user', {
  state: (): UserState => ({
    token: localStorage.getItem('token'),
    userInfo: null,
    tenantId: null,
    role: null
  }),

  getters: {
    isLoggedIn: (state) => !!state.token,
    isPlatformAdmin: (state) => state.role === 'platform_admin',
    isShopAdmin: (state) => state.role === 'shop_admin',
    isShopStaff: (state) => state.role === 'shop_staff'
  },

  actions: {
    async login(username: string, password: string, tenantId?: number) {
      const res = await loginApi({ username, password, tenantId })
      this.token = res.token
      this.userInfo = res.admin
      this.tenantId = tenantId || null
      this.role = res.admin.role
      localStorage.setItem('token', res.token)
    },

    async fetchCurrentUser() {
      const res = await getCurrentUser()
      this.userInfo = res
    },

    logout() {
      this.token = null
      this.userInfo = null
      this.tenantId = null
      this.role = null
      localStorage.removeItem('token')
    }
  }
})
```

### 应用状态 (app.ts)

```typescript
// stores/app.ts
import { defineStore } from 'pinia'

interface AppState {
  sidebarCollapsed: boolean
  loading: boolean
}

export const useAppStore = defineStore('app', {
  state: (): AppState => ({
    sidebarCollapsed: false,
    loading: false
  }),

  actions: {
    toggleSidebar() {
      this.sidebarCollapsed = !this.sidebarCollapsed
    },
    setLoading(loading: boolean) {
      this.loading = loading
    }
  }
})
```

### 门店配置状态 (shop.ts)

```typescript
// stores/shop.ts
import { defineStore } from 'pinia'
import { getShopConfig } from '@/api/shop'

interface ShopState {
  shopConfig: any | null
}

export const useShopStore = defineStore('shop', {
  state: (): ShopState => ({
    shopConfig: null
  }),

  actions: {
    async fetchShopConfig() {
      const res = await getShopConfig()
      this.shopConfig = res
    }
  }
})
```
